【重新开始】功能：

在 js 文件里，ctrl+F 查找 update，在后面加上一行 location = location 

update() {
    const {
      checkerBoard,
      corsor
    } = this;
    checkerBoard.draw();
    corsor.draw();
    if (this.isGameOver()) {
      alert(this.switchPlayer ? '黑方胜利' : '白方胜利');
      //刷新页面重新开始
      location = location
    }
  }

2、在 game.js 里面 ctrl+F 查找 self.hisAction.push，这是一个存储操作记录的数组，在他的 push 方法下面的空白区操作（我做的可能有点麻烦） 

self.hisAction.push({ type: 'CHEKCER_DOWN', value: { x: x, y: y } });
self.switchPlayer = !self.switchPlayer;
self.update(); //更新游戏状态

//显示操作记录
let down = self.hisAction.length; //.length获取落子个数
let cx = self.hisAction[down - 1].value.x; //获取x坐标
let cy = self.hisAction[down - 1].value.y; //获取y坐标
let color; //判断落子数。是奇数的话就是黑色
down % 2 == 1 ? (color = '黑') : (color = '白');
let ul = document.querySelector('ul'); //获取ul
let li;
li = document.createElement('li'); //创建li
//在ul的每一个子节点前面添加li
ul.insertBefore(li, ul.children[0]); //新发送的，永远在第一行
//li的内容：
li.innerHTML = '第' + down + '子,' + color + '子,坐标:' + cx + ',' + cy;
