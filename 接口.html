MOV DX,290H            ;送端口地址
L1: MOV AL,0               ;最低点开始，送0
L2: OUT DX,AL
CALL DELAY                  ;延时
ADD AL,16                    ;按步长增加
JNZ L2                           ;加到256为0停止
MOV AL.0FFH                  ;取不到256最高点强制变255
OUT DX,AL                       ;输出
CALL DELAY                   ;延时  上升结束
MOV AL,0F0H             ; 开始下降 从256开始下降（否则最后点不为0）取240
L3: OUT DX,AL
CALL DELAY               ;延时
SUB LA,16                ;按步长下降
JNZ L3                   ;降到0停止
JMP L1                  ;强制跳转
